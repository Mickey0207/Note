```python
my_list = ["my_list",1,2,3,4,5,6,]

print(max(my_list[1:8]))     #取得串列的最大值
print(min(my_list[1:8]))     #取得串列的最小值
print(sum(my_list[1:8]))     #取得串列的總和

print(len(my_list))      #取得串列中元素的個數

my_list[0] = "ttt"

print(my_list)
```
